293  template< typename DATA >
294  void buffer::load_attribute( GLuint index, std::vector< DATA > const& attrib_data )
295  {
296  if ( not verts_specified ) {
297  throw std::logic_error( "Attribute value assignment attempted when vertex format was not specified." );
298  }
299  if ( (*(*attributes)[index]) != type< DATA >() ) {
300  std::string msg = "Type stored in std::vector, ";
301  msg += type< DATA >().name();
302  msg += ", does not match type specified at buffer index ";
303  msg += index;
304  msg += ", ";
305  msg += (*attributes)[index]->name();
306  msg += ".";
307  throw std::invalid_argument( msg );
308  }
309  if ( attrib_data.size() > n_blocks ) {
310  std::string msg = "Attribute value assignment attempted with ";
311  msg += n_blocks;
312  msg += " allocated data blocks but ";
313  msg += attrib_data.size();
314  msg += " attribute values.";
315  throw std::invalid_argument( msg);
316  }
317 
318  if ( data == 0 ) {
319  data = new unsigned char[n_blocks * stride];
320  }
321 
322  GLsizeiptr block;
323  unsigned char* cursor = data;
324  cursor += attribute_offset( index );
325  for( block = 0; block < n_blocks; ++block ) {
326  raw_map const mapped_data = attrib_data[block].to_map();
327  for( size_t b = 0; b < mapped_data.n_bytes; ++b ) {
328  cursor[b] = mapped_data[b];
329  }
330  cursor += stride;
331  }
332  }
