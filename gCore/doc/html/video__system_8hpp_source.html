258  inline void video_system::unregister_context( context* cntx )
259  {
260  context_set::iterator e = contexts->find( cntx );
261  if ( e != contexts->end() ) {
262  // The context IS in this set
263  if ( cntx == active_context ) {
264  // If it is also the active context, we need to reseat
265  // the active context
266  if ( contexts->size() == 1 ) {
267  // If there is only one context, then the pointer must
268  // be assigned null
269  active_context = 0;
270  } else if ( e == --(contexts->end()) ) {
271  // If the context is the last in the set, then we need
272  // need to back track
273  active_context = *(--e);
274  ++e;
275  } else {
276  // Otherwise, we can forward track
277  active_context = *(++e);
278  --e;
279  }
280  }
281  // Now we can erase the pointer from the set
282  contexts->erase( e );
283  }
